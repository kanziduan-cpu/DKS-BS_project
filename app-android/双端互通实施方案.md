## 一、服务器架构规划

### 1.1 需要部署的服务

#### A. REST API服务 (端口8000)
- **功能**: 处理HTTP API请求
- **技术栈**: Node.js / Python Flask / Java Spring Boot
- **端口**: 8000
- **作用**: 用户认证、仓库管理、设备控制、数据查询

#### B. MQTT Broker (端口1883)
- **功能**: 实时消息通信
- **技术栈**: EMQX / Mosquitto
- **端口**: 1883 (MQTT), 18083 (管理面板)
- **作用**: 
  - 设备端→云端: 上报环境数据、设备状态
  - 云端→移动端: 推送实时数据、报警信息
  - 移动端→云端: 发送控制指令

#### C. 数据库服务
- **功能**: 数据持久化存储
- **技术栈**: MySQL / MongoDB / PostgreSQL
- **端口**: 3306 (MySQL) / 27017 (MongoDB)
- **作用**: 存储用户数据、仓库信息、历史数据、报警记录

## 二、双端通信架构

### 2.1 端到端数据流

```
[单片机设备端]
    ↓ MQTT (1883)
[MQTT Broker - EMQX]
    ↓ MQTT (1883)
[移动端 Android App]
    ↓ HTTP (8000)
[REST API服务]
    ↓
[数据库]
```

### 2.2 详细数据流

#### 数据上报流程 (设备端 → 移动端)
1. 单片机通过MQTT连接到EMQX (1883端口)
2. 单片机发布环境数据到主题: `test/environment`
3. EMQX转发消息到所有订阅者
4. Android App订阅 `test/#` 主题
5. App收到数据后更新UI显示

#### 设备控制流程 (移动端 → 设备端)
1. 用户在App点击设备控制按钮
2. App通过MQTT发布控制指令到: `test/device/control`
3. 单片机订阅 `test/device/control` 主题
4. 单片机接收指令并控制硬件(风扇、水泵等)
5. 单片机发布新的设备状态到: `test/device/status`

#### 用户认证流程 (App → API服务)
1. 用户在App输入用户名密码
2. App通过HTTP POST到 `/api/auth/login`
3. REST API验证用户信息
4. 返回Token和用户数据
5. App保存Token用于后续请求

### 3.2 部署MQTT Broker (EMQX)

#### 方案A: 使用Docker部署EMQX (推荐)

创建配置文件 `docker-compose.yml`:
```yaml
version: '3.8'

services:
  emqx:
    image: emqx/emqx:5.3.0
    container_name: emqx_server
    ports:
      - "1883:1883"      # MQTT协议
      - "8083:8083"      # MQTT over WebSocket
      - "8883:8883"      # MQTT over SSL
      - "18083:18083"    # Dashboard管理界面
    environment:
      - EMQX_NAME=emqx
      - EMQX_HOST=0.0.0.0
      - EMQX_DASHBOARD__DEFAULT__PASSWORD=public123
    volumes:
      - emqx_data:/opt/emqx/data
      - emqx_log:/opt/emqx/log
    restart: unless-stopped
    networks:
      - warehouse_net

  mysql:
    image: mysql:8.0
    container_name: mysql_server
    ports:
      - "3306:3306"
    environment:
      - MYSQL_ROOT_PASSWORD=warehouse123
      - MYSQL_DATABASE=warehouse_db
      - MYSQL_USER=warehouse_user
      - MYSQL_PASSWORD=warehouse_pass
    volumes:
      - mysql_data:/var/lib/mysql
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    restart: unless-stopped
    networks:
      - warehouse_net

  api_server:
    build: ./api
    container_name: api_server
    ports:
      - "8000:8000"
    environment:
      - DB_HOST=mysql
      - DB_PORT=3306
      - DB_USER=warehouse_user
      - DB_PASSWORD=warehouse_pass
      - DB_NAME=warehouse_db
      - MQTT_HOST=emqx
      - MQTT_PORT=1883
      - MQTT_USER=testuser
      - MQTT_PASSWORD=123456
    depends_on:
      - mysql
      - emqx
    restart: unless-stopped
    networks:
      - warehouse_net

volumes:
  emqx_data:
  emqx_log:
  mysql_data:

networks:
  warehouse_net:
    driver: bridge
```

### 3.4 部署REST API服务

#### 创建API服务代码 (Node.js版本)

在 `api/` 目录下创建文件:

**package.json**:
```json
{
  "name": "warehouse-api",
  "version": "1.0.0",
  "main": "server.js",
  "dependencies": {
    "express": "^4.18.2",
    "mysql2": "^3.6.0",
    "mqtt": "^5.1.3",
    "cors": "^2.8.5",
    "bcryptjs": "^2.4.3",
    "jsonwebtoken": "^9.0.2",
    "dotenv": "^16.3.1"
  }
}
```

**Dockerfile**:
```dockerfile
FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
EXPOSE 8000
CMD ["node", "server.js"]
```

**server.js**:
```javascript
const express = require('express');
const mysql = require('mysql2/promise');
const mqtt = require('mqtt');
const cors = require('cors');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

const app = express();
const PORT = 8000;

// 中间件
app.use(cors());
app.use(express.json());

// 数据库配置
const dbConfig = {
  host: process.env.DB_HOST || 'mysql',
  port: process.env.DB_PORT || 3306,
  user: process.env.DB_USER || 'warehouse_user',
  password: process.env.DB_PASSWORD || 'warehouse_pass',
  database: process.env.DB_NAME || 'warehouse_db',
  waitForConnections: true,
  connectionLimit: 10
};

const pool = mysql.createPool(dbConfig);

// MQTT配置
const mqttConfig = {
  host: process.env.MQTT_HOST || 'emqx',
  port: process.env.MQTT_PORT || 1883,
  username: process.env.MQTT_USER || 'testuser',
  password: process.env.MQTT_PASSWORD || '123456'
};

const mqttClient = mqtt.connect(`mqtt://${mqttConfig.host}:${mqttConfig.port}`, {
  username: mqttConfig.username,
  password: mqttConfig.password
});

mqttClient.on('connect', () => {
  console.log('MQTT Connected to broker');
  mqttClient.subscribe('test/device/status');
  mqttClient.subscribe('test/environment');
});

mqttClient.on('message', (topic, message) => {
  console.log(`Received message on ${topic}: ${message.toString()}`);
  // 可以在这里处理MQTT消息并存入数据库
});

// API路由

// 用户登录
app.post('/api/auth/login', async (req, res) => {
  try {
    const { username, password } = req.body;
    
    const [rows] = await pool.query(
      'SELECT * FROM users WHERE username = ?',
      [username]
    );
    
    if (rows.length === 0) {
      return res.json({ success: false, message: '用户不存在' });
    }
    
    const user = rows[0];
    const isValid = await bcrypt.compare(password, user.password);
    
    if (!isValid) {
      return res.json({ success: false, message: '密码错误' });
    }
    
    const token = jwt.sign({ userId: user.id }, 'secret_key', { expiresIn: '24h' });
    
    res.json({
      success: true,
      message: '登录成功',
      token,
      user: {
        id: user.id,
        username: user.username,
        nickname: user.nickname
      }
    });
  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({ success: false, message: '服务器错误' });
  }
});

// 获取仓库列表
app.get('/api/warehouse/list', async (req, res) => {
  try {
    const { userId } = req.query;
    
    const [rows] = await pool.query(`
      SELECT w.* FROM warehouses w
      JOIN user_warehouses uw ON w.id = uw.warehouse_id
      WHERE uw.user_id = ?
    `, [userId]);
    
    res.json(rows);
  } catch (error) {
    console.error('Get warehouses error:', error);
    res.status(500).json({ message: '服务器错误' });
  }
});

// 获取设备列表
app.get('/api/device/list', async (req, res) => {
  try {
    const { warehouseId } = req.query;
    
    const [rows] = await pool.query(
      'SELECT * FROM devices WHERE warehouse_id = ?',
      [warehouseId]
    );
    
    res.json(rows);
  } catch (error) {
    console.error('Get devices error:', error);
    res.status(500).json({ message: '服务器错误' });
  }
});

// 设备控制
app.post('/api/device/control', (req, res) => {
  try {
    const { deviceId, action, value } = req.body;
    
    const message = JSON.stringify({
      deviceId,
      action,
      value,
      timestamp: Date.now()
    });
    
    mqttClient.publish('test/device/control', message);
    
    res.json({ success: true, message: '控制指令已发送' });
  } catch (error) {
    console.error('Control device error:', error);
    res.status(500).json({ success: false, message: '服务器错误' });
  }
});

// 获取当前环境数据
app.get('/api/environment/current', async (req, res) => {
  try {
    const { warehouseId } = req.query;
    
    const [rows] = await pool.query(`
      SELECT * FROM environment_data 
      WHERE warehouse_id = ? 
      ORDER BY timestamp DESC 
      LIMIT 1
    `, [warehouseId]);
    
    if (rows.length > 0) {
      res.json(rows[0]);
    } else {
      res.json({
        temperature: 0,
        humidity: 0,
        formaldehyde: 0,
        co: 0,
        co2: 0,
        aqi: 0,
        timestamp: Date.now()
      });
    }
  } catch (error) {
    console.error('Get current data error:', error);
    res.status(500).json({ message: '服务器错误' });
  }
});

// 获取历史数据
app.get('/api/environment/history', async (req, res) => {
  try {
    const { warehouseId, startTime, endTime, paramType } = req.query;
    
    const [rows] = await pool.query(`
      SELECT timestamp, ${paramType} as value 
      FROM environment_data 
      WHERE warehouse_id = ? 
      AND timestamp BETWEEN ? AND ?
      ORDER BY timestamp ASC
    `, [warehouseId, startTime, endTime]);
    
    res.json(rows);
  } catch (error) {
    console.error('Get history data error:', error);
    res.status(500).json({ message: '服务器错误' });
  }
});

// 获取报警列表
app.get('/api/alarm/list', async (req, res) => {
  try {
    const { warehouseId } = req.query;
    
    const [rows] = await pool.query(`
      SELECT * FROM alarms 
      WHERE warehouse_id = ? 
      ORDER BY timestamp DESC
    `, [warehouseId]);
    
    res.json(rows);
  } catch (error) {
    console.error('Get alarms error:', error);
    res.status(500).json({ message: '服务器错误' });
  }
});

// 健康检查
app.get('/api/health', (req, res) => {
  res.json({ status: 'ok', timestamp: Date.now() });
});

// 启动服务器
app.listen(PORT, '0.0.0.0', () => {
  console.log(`API Server running on port ${PORT}`);
});
```

## 五、设备端配置(单片机)

### 5.1 MQTT连接配置(C/STM32)

```c
#include <string.h>
#include "mqtt_client.h"

// MQTT配置
#define MQTT_BROKER "43.99.24.178"  // 或您的服务器IP
#define MQTT_PORT 1883
#define MQTT_USER "testuser"
#define MQTT_PASS "123456"
#define MQTT_CLIENT_ID "stm32_device_001"

// 主题定义
#define TOPIC_ENV "test/environment"
#define TOPIC_STATUS "test/device/status"
#define TOPIC_CONTROL "test/device/control"

// MQTT客户端结构
MQTTClient client;

// 初始化MQTT
void init_mqtt() {
    MQTTClient_init(&client, MQTT_BROKER, MQTT_PORT, MQTT_USER, MQTT_PASS, MQTT_CLIENT_ID);
}

// 发送环境数据
void send_environment_data(float temp, float hum, float co2) {
    char payload[256];
    snprintf(payload, sizeof(payload),
        "{\"temperature\":%.2f,\"humidity\":%.2f,\"co2\":%.2f,\"timestamp\":%ld}",
        temp, hum, co2, (long)time(NULL));
    
    MQTTClient_publish(&client, TOPIC_ENV, payload);
}

// 接收控制指令回调
void on_control_message(const char* topic, const char* payload) {
    // 解析JSON并控制硬件
    // 例如: {"deviceId":"device-fan-001","action":"turn_on","value":"high"}
    
    if (strstr(payload, "turn_on")) {
        // 开启设备
    } else if (strstr(payload, "turn_off")) {
        // 关闭设备
    }
}
```
